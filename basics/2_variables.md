

# Operators


# Math operators
+ = addition
- = substract
/ = divide
* = multiply
% = modulus

// example of x to power of (pow):
let squared = i32::pow(8, 2)

// float to power of integer (powi)
let float_integer = f32::powi(8.1, 2);

// float to power of float (powf)
let float_float = f32::powf(8.1, 2.2);


# Bitwise operators

& = and
|  = or
^  = and-or
<< = shift
>> = shift
